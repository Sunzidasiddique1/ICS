import tkinter as tk
from tkinter import messagebox
import random

class HMIApp(tk.Tk):
    def __init__(self):
        super().__init__()
        self.title("HMI for Industrial Control System")

        self.sensor_label = tk.Label(self, text="Sensor Data:", font=("Arial", 14))
        self.sensor_label.pack(pady=10)

        self.sensor_display = tk.Label(self, text="0.0", font=("Arial", 24))
        self.sensor_display.pack(pady=10)

        self.start_button = tk.Button(self, text="Start Process", font=("Arial", 14), command=self.start_process)
        self.start_button.pack(pady=10)

        self.stop_button = tk.Button(self, text="Stop Process", font=("Arial", 14), command=self.stop_process)
        self.stop_button.pack(pady=10)

        self.update_sensor_data()

    def start_process(self):
        messagebox.showinfo("Process Control", "Process Started")
        print("Process started.")

    def stop_process(self):
        messagebox.showinfo("Process Control", "Process Stopped")
        print("Process stopped.")

    def update_sensor_data(self):
        sensor_value = random.uniform(20.0, 30.0)
        self.sensor_display.config(text=f"{sensor_value:.2f}")
        self.after(1000, self.update_sensor_data)

if __name__ == "__main__":
    app = HMIApp()
    app.mainloop()
